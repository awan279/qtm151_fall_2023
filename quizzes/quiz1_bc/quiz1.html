# Write your own code:
list_names = ["data_quiz/wdi_agedist_1980.csv","data_quiz/wdi_agedist_2000.csv","data_quiz/wdi_agedist_2020.csv"]
index = 0

for name in list_names:
    dataset = pd.read_csv(name)
    percent_ages0to14 = dataset["percent_ages0to14"]
    percent_ages15orolder = 100 - percent_ages0to14
    plt.hist(percent_ages15orolder)
    plt.title("Histogram of Dataset")
    plt.ylabel("Frequency")
    plt.xlabel("Proportion that is 15 or older")
    plt.show()
    index = index + 1
